\section{Overview}

\textbf{Yet Another Audio analyzeR (YAAR)} is a real-time audio measurement and analysis tool designed for:

\begin{itemize}
    \item Audio signal inspection (time and frequency domain)
    \item FFT-based spectral analysis
    \item Distortion analysis (THD, THD+N, IMD)
    \item Signal quality metrics (SNR, SINAD, ENOB)
    \item Power calculations (Vrms, Prms)
    \item Signal simulation for testing
\end{itemize}

\section{Features}

\subsection{Core Capabilities}
\begin{itemize}
    \item Real-time FFT visualization
    \item Rolling FFT averaging
    \item Automatic fundamental detection
    \item THD / THD+N computation
    \item IMD (two-tone intermodulation distortion)
    \item Noise floor estimation
\end{itemize}

\subsection{Output}
\begin{itemize}
    \item CSV logging
    \item Plot export
\end{itemize}

    \item Shows fundamental tones and harmonics
\end{itemize}

\subsection{Status Panel}

    \item Frequency (F1, F2)
    \item THD / THD+N or IMD
    \item SNR, ENOB
    \item Voltage and power
\end{itemize}

\section{Measurement Metrics}

\subsection{THD}
Total harmonic distortion:
\[
\mathrm{THD} = \sqrt{\frac{V_{\mathrm{dist}}}{V_{\mathrm{fund}}}}
\]

\subsection{THD+N}
Includes noise:
\[
\mathrm{THD+N} = \sqrt{\frac{V_{\mathrm{dist}} + V_{\mathrm{noise}}}{V_{\mathrm{fund}}}}
\]

\subsection{SNR}
\[
\mathrm{SNR} = 20 \log_{10} \left(\sqrt{\frac{V_{\mathrm{fund}}}{V_{\mathrm{noise}}}} \right)
\]

\subsection{ENOB}
\[
\mathrm{ENOB} = \frac{\mathrm{SINAD} - 1.76}{6.02}
\]

\section{Operating Modes}

\subsection{Single-Tone Mode}
\begin{itemize}
    \item One dominant frequency
    \item Computes THD and THD+N
\end{itemize}

\subsection{Two-Tone Mode}
\begin{itemize}
    \item Two detected tones
    \item Computes IMD and CCIF IMD
\end{itemize}

\section{Tips for Accurate Measurements}

\begin{itemize}
    \item Use large FFT sizes
    \item Avoid clipping
    \item Match signal frequency to FFT bins
    \item Ensure low-noise input
\end{itemize}

\section{Troubleshooting}

\subsection{No signal}
\begin{itemize}
    \item Check device ID
    \item Verify channel selection
\end{itemize}

\subsection{Noisy FFT}
\begin{itemize}
    \item Increase FFT size
\end{itemize}

\section{Limitations}

\begin{itemize}
    \item Requires external audio backend
    \item CLI-only interface
    \item Performance depends on CPU
\end{itemize}

\section{Exit Conditions}

The program stops when:
\begin{itemize}
    \item Duration expires
    \item Window is closed
    \item Ctrl+C is pressed
\end{itemize}
